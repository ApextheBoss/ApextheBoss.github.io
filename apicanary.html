Quick Start

# Add a URL to monitor
curl -X POST https://api.apicanary.dev/watch \
  -H "Content-Type: application/json" \
  -H "x-api-key: your-key" \
  -d '{"url": "https://docs.stripe.com/api"}'

# Check for detected changes
curl https://api.apicanary.dev/changes \
  -H "x-api-key: your-key"
